type pagination[T any] struct {
	data    []T
	perPage uint
}

// start form 1
func (p pagination[T]) getPage(i uint) []T { // tar buat ambil datanya
	start := p.perPage * (i - 1)
	if start > uint(len(p.data)) {
		return nil
	}
	end := min(start+p.perPage, uint(len(p.data)))
	return p.data[start:end]
}

func (p pagination[T]) getTotalPages() uint { // tar buat tampilin maxPage (label dibawah table, [1..maxPage])
	return uint(math.Ceil(float64(len(p.data)) / float64(p.perPage)))
}

func newPagination[T any](data []T, perPage uint) pagination[T] {
	return pagination[T]{data, perPage}
}
